# Monitor Riser Speakers – Concept Summary

import Speaker from '@pocs/3d-speaker-viewer/Viewer'

export const speakerWidth = 19
export const speakerHeight = 19
export const speakerDepth = 30
export const volume = (speakerWidth * speakerHeight * speakerDepth) / 1000

export const panelThickness = 2.1
export const internalWidth = speakerWidth - 2 * panelThickness
export const internalHeight = speakerHeight - 2 * panelThickness
export const internalDepth = speakerDepth - 2 * panelThickness
export const internalVolume = (internalWidth * internalHeight * internalDepth) / 1000

export const plateArea_m2 = (internalWidth / 100) * (internalDepth / 100)
export const plateMassKg = 3
export const steelDensity = 7850
export const graniteDensity = 2700
export const leadDensity = 11340
export const castIronDensity = 7200

export const requiredThicknessSteel_m = plateMassKg / (steelDensity * plateArea_m2)
export const requiredThicknessGranite_m = plateMassKg / (graniteDensity * plateArea_m2)
export const requiredThicknessSteel_cm = requiredThicknessSteel_m * 100
export const requiredThicknessGranite_cm = requiredThicknessGranite_m * 100

export const steelE = 210e9
export const stoneE = 50e9
export const leadE = 16e9
export const castIronE = 120e9
export const poissonSteel = 0.3
export const poissonStone = 0.25
export const poissonLead = 0.44
export const poissonCastIron = 0.27

export const tSteel = requiredThicknessSteel_m
export const tGranite = requiredThicknessGranite_m
export const tLead = plateMassKg / (leadDensity * plateArea_m2)
export const tCastIron = plateMassKg / (castIronDensity * plateArea_m2)

export const D_steel = steelE * Math.pow(tSteel, 3) / (12 * (1 - Math.pow(poissonSteel, 2)))
export const D_granite = stoneE * Math.pow(tGranite, 3) / (12 * (1 - Math.pow(poissonStone, 2)))
export const D_lead = leadE * Math.pow(tLead, 3) / (12 * (1 - Math.pow(poissonLead, 2)))
export const D_castIron = castIronE * Math.pow(tCastIron, 3) / (12 * (1 - Math.pow(poissonCastIron, 2)))

export const mPerArea_steel = steelDensity * tSteel
export const mPerArea_granite = graniteDensity * tGranite
export const mPerArea_lead = leadDensity * tLead
export const mPerArea_castIron = castIronDensity * tCastIron

export const a_m = Math.max(internalWidth, internalDepth) / 100
export const C = 1.0
export const fFund_steel = C * Math.sqrt(D_steel / (mPerArea_steel * Math.pow(a_m, 4)))
export const fFund_granite = C * Math.sqrt(D_granite / (mPerArea_granite * Math.pow(a_m, 4)))
export const fFund_lead = C * Math.sqrt(D_lead / (mPerArea_lead * Math.pow(a_m, 4)))
export const fFund_castIron = C * Math.sqrt(D_castIron / (mPerArea_castIron * Math.pow(a_m, 4)))

export const fFund_steel_clamped = fFund_steel * 1.6
export const fFund_granite_clamped = fFund_granite * 1.6
export const fFund_lead_clamped = fFund_lead * 1.6
export const fFund_castIron_clamped = fFund_castIron * 1.6

export const acousticRatingSteel = 3
export const acousticRatingGranite = 5
export const acousticRatingLead = 8
export const acousticRatingCastIron = 6

export const neopreneIsolationFraction = 0.5

export function fundamentalsScore(f) {
  if (f < 40) return 40
  if (f < 200) return 20
  return 0
}

export const combinedScoreSteel = Math.round((acousticRatingSteel / 10) * 30 + fundamentalsScore(fFund_steel) + neopreneIsolationFraction * 30)
export const combinedScoreLead = Math.round((acousticRatingLead / 10) * 30 + fundamentalsScore(fFund_lead) + neopreneIsolationFraction * 30)
export const combinedScoreCastIron = Math.round((acousticRatingCastIron / 10) * 30 + fundamentalsScore(fFund_castIron) + neopreneIsolationFraction * 30)
export const combinedScoreGranite = Math.round((acousticRatingGranite / 10) * 30 + fundamentalsScore(fFund_granite) + neopreneIsolationFraction * 30)

export function scoreFor(name) {
  switch (name) {
    case 'Steel': return combinedScoreSteel
    case 'Lead': return combinedScoreLead
    case 'CastIron': return combinedScoreCastIron
    case 'Granite': return combinedScoreGranite
    default: return 0
  }
}

export function densityFor(name) {
  switch (name) {
    case 'Steel': return steelDensity
    case 'Lead': return leadDensity
    case 'CastIron': return castIronDensity
    case 'Granite': return graniteDensity
    default: return 0
  }
}

export function thicknessCmFor(name) {
  switch (name) {
    case 'Steel': return requiredThicknessSteel_cm
    case 'Lead': return tLead * 100
    case 'CastIron': return tCastIron * 100
    case 'Granite': return requiredThicknessGranite_cm
    default: return 0
  }
}

export function notesFor(name) {
  switch (name) {
    case 'Steel': return 'Most compact; easy to machine and secure.'
    case 'Lead': return 'Very dense — thin sheets or inserts; toxic dust when cut; handle carefully.'
    case 'CastIron': return 'Dense and workable cast material; heavier than steel for same thickness.'
    case 'Granite': return 'Thicker; premium look if exposed; heavier to handle.'
    default: return ''
  }
}

export function acousticPropsFor(name) {
  switch (name) {
    case 'Steel': return 'Low inherent damping; reflects structure-borne vibration and may ring if not isolated.'
    case 'Lead': return 'Highly damped internally and excellent at absorbing vibration when isolated; handle safely.'
    case 'CastIron': return 'Cast iron has moderate damping and good mass; performs well for ballast.'
    case 'Granite': return 'Moderate damping; transmits low frequencies efficiently; brittle, secure mounting needed.'
    default: return ''
  }
}

export function fFundFor(name) {
  switch (name) {
    case 'Steel': return fFund_steel
    case 'Lead': return fFund_lead
    case 'CastIron': return fFund_castIron
    case 'Granite': return fFund_granite
    default: return 0
  }
}

export function fFundClampedFor(name) {
  switch (name) {
    case 'Steel': return fFund_steel_clamped
    case 'Lead': return fFund_lead_clamped
    case 'CastIron': return fFund_castIron_clamped
    case 'Granite': return fFund_granite_clamped
    default: return 0
  }
}

export function riskBandFor(name) {
  const f = fFundFor(name)
  if (f < 60) return 'Low'
  if (f < 200) return 'Medium'
  return 'High'
}

export function combinedScoreFor(name) {
  return scoreFor(name)
}

export function displayName(name) {
  if (name === 'CastIron') return 'Cast Iron'
  return name
}

export const materialOrder = ['Steel', 'Lead', 'CastIron', 'Granite'].slice().sort((a, b) => scoreFor(b) - scoreFor(a))

export function acousticPointsFor(name) {
  switch (name) {
    case 'Steel': return (acousticRatingSteel / 10) * 30
    case 'Lead': return (acousticRatingLead / 10) * 30
    case 'CastIron': return (acousticRatingCastIron / 10) * 30
    case 'Granite': return (acousticRatingGranite / 10) * 30
    default: return 0
  }
}

export function fundamentalsPointsFor(name) {
  return fundamentalsScore(fFundFor(name))
}

export function neoprenePointsFor(name) {
  return neopreneIsolationFraction * 30

## Background

The project started with a **[custom monitor riser](https://tobias-belch.github.io/ideas/docs/monitor-riser/aorus-fv43u-monitor-riser/)** design for a large display (AORUS FV43U).  
While structurally sound, the resulting risers were **bulky** and consumed a significant amount of desk space.

To compensate and add value, the idea evolved into turning the risers into **fully functional desk speakers**.

## Core Idea

Transform each monitor riser into a **stereo speaker enclosure**, creating:

- A **space-efficient audio solution**
- Proper speaker placement for near-field listening
- A multi-functional piece of desk furniture

Each riser becomes part of a **left/right stereo pair**, with one acting as a *master* (electronics) and the other as a *slave* (speaker only).

## Target Dimensions

Per riser (external):

- **Width (X):** {speakerWidth} cm  
- **Height (Y):** {speakerHeight} cm  
- **Depth (Z):** {speakerDepth} cm  

These outside dimensions provide approximately **{volume.toFixed(1)} liters of external volume**. With a uniform panel thickness of **{panelThickness * 10} cm** the internal usable acoustic volume becomes approximately **{internalVolume.toFixed(1)} liters** (before accounting for braces, driver displacement and internal compartments), which is in the target range for compact desktop speakers.

## Feasibility Assessment

### DIY vs Off-the-Shelf

- Existing commercial speakers do **not fit the dimensions well**
- DIY is **not only possible, but preferable**
- All required electronics exist as **off-the-shelf modules**

### Functional Requirements

The speaker system supports:

- 3.5 mm analog audio input
- Bluetooth audio
- Stereo amplification
- Optional headphone output with speaker muting
- Front-accessible controls

No custom PCBs or DSP are required.

## Enclosure Design Principles

- **Front-firing driver and bass port**
- Optimised for near-wall and desk placement
- Internally braced to handle:
  - Acoustic pressure
  - Monitor weight
- Acoustic damping inside the enclosure
- Electronics isolated from the speaker air volume

## Materials & Construction

- **Primary material:** 18–21 mm plywood or MDF (birch plywood preferred for visible surfaces). Use 18 mm for side panels and 21 mm for the top plate if the top will contain the foot-lock pocket.
- **Top plate / load spreader:** double-thickness top (two 18 mm layers laminated, or single 21–24 mm board) where the monitor feet sit. Reinforce the foot-pocket area with a small doubler plate beneath the top layer.
- **Internal bracing material:** same plywood as the enclosure. All braces should be glued and screwed; use dovetail or lap joints where possible for rigidity.
- **Bottom ballast:** use a steel plate sized to the internal floor (approximately **{internalWidth.toFixed(1)} × {internalDepth.toFixed(1)} cm**) with thickness **{requiredThicknessSteel_cm} cm**. A plate of this footprint and thickness has approximately **{plateMassKg.toFixed(2)} kg** mass per enclosure; place it in a dedicated tray at the very bottom. Use two plates (one per enclosure) for the stereo pair.

### Note on fundamental frequency

- **Why it matters:** the plate's fundamental (lowest) vibration frequency determines where the ballast can happily coexist with the speaker system. If the plate's resonance falls inside the speaker's passband (or near strong driver output peaks), the ballast can be excited and produce audible ringing or colouration, or couple structure-borne vibration into the desk and room.
- **What influences it:** plate size and aspect ratio, thickness, material stiffness (Young's modulus), mass per unit area, and edge support (clamped vs simply-supported). Heavier and stiffer plates lower or raise resonant frequencies depending on thickness vs mass distribution; increasing mass per area tends to lower modal frequencies, while increasing stiffness (t³ dependence on thickness) raises them.
- **Good vs bad values:**
  - Good: fundamental frequencies well below the speaker's lowest useful output (e.g., < 40 Hz for small desktop speakers) or well above the audible bass band so they are not excited by normal playback. Plate fundamentals in the 20–40 Hz range are typically safe for small desktop speakers because most energy above that is lower. Higher plate fundamentals (several hundred Hz) are risky because they sit inside the critical mid/bass band and will be easily excited.
  - Bad: plate fundamentals that coincide with driver resonance, bass port tuning, or strong musical content (e.g., 60–300 Hz). These will produce perceptible ringing, boominess, or unwanted vibrations transmitted to the desk.
- **Mitigations:** use Sorbothane or rubber isolation between plate and enclosure to damp structure-borne energy; increase mass (thicker plate or denser material) to move fundamentals lower; add constrained-layer damping; ensure plate is well supported and constrained to reduce Q and spread modal energy.

- **Table row explanations (short):**
  - **Attachment (thin rubber / neoprene pads):** shows the recommended practical decoupling solution. The numeric value shown is an estimated insertion-loss percentage (and approximate dB) for a thin neoprene pad — it does not change the plate's natural frequency much, but reduces how strongly that mode is excited and how much energy transmits into the desk.
  - **Risk band:** a simple classification (Low / Medium / High) derived from the plate fundamental relative to typical audio bands. "Low" means the fundamental is unlikely to be excited by normal playback; "High" means the fundamental could sit inside mid/bass content and cause audible issues.
  - **Neoprene isolation (est.):** an approximate, frequency‑averaged figure expressing how effective the neoprene pads are at reducing transmitted vibration. It's a rule-of-thumb — real insertion loss depends on pad stiffness, contact area, compression and frequency.
  - **Combined acoustic score (0–100):** a heuristic aggregate for quick comparison combining material damping, where the fundamental lies, and the assumed neoprene isolation. Use it only for relative ranking — it's not a substitute for measurements or FEA.

### Recommendation

- **Primary choice — Steel plate (practical, compact):** use a steel plate sized to the internal floor ({internalWidth.toFixed(1)} × {internalDepth.toFixed(1)} cm) with thickness **{requiredThicknessSteel_cm.toFixed(2)} cm** (≈ {plateMassKg.toFixed(2)} kg). Steel is easy to machine, predictable under load, and economical. Install it in a shallow tray at the enclosure bottom.
- **Attachment / isolation (required):** decouple the plate from the desk using thin neoprene pads (3–5 mm) or a continuous neoprene/Sorbothane sheet under the enclosure. This gives the practical balance of decoupling without dramatically changing the plate fundamentals. Use 3–4 small pads placed near the enclosure corners if a continuous sheet is impractical.
- **Preferred high-acoustic option — Lead insert (best acoustic score):** if you can source pre-cut lead sheets or safe lead-lined inserts, a thin lead sheet achieves the same mass with much smaller thickness (**{(tLead*100).toFixed(2)} cm**) and gives lower modal frequency and high internal damping. Handle lead safely (wear PPE while cutting, avoid dust) and encapsulate it so it cannot be scratched or shed.
- **Alternative — Cast iron:** if lead is not acceptable, cast iron provides good mass and moderate damping with thickness **{(tCastIron*100).toFixed(2)} cm**. Cast iron is brittle — avoid sharp impacts and secure it well.
- **Aesthetic option — Stone (granite):** granite looks premium but requires significantly more thickness (**{requiredThicknessGranite_cm.toFixed(2)} cm**) and careful handling and mounting to avoid chipping.
- **Safety & installation notes:**
  - When cutting or machining lead, use respirators and wet-cut methods; encapsulate or coat the finished element. Do not sand or grind lead on-site without proper controls.
  - Use fixings that clamp the ballast into a tray; avoid relying on glue alone for heavy plates.
  - Add a thin constrained-layer damping pad (adhesive rubber + bitumen or a Sorbothane washer) near high-stress contact points if residual ringing is audible after neoprene isolation.

This recommendation reflects the numeric scoring and modal analysis above: steel is the pragmatic choice, while lead (or lead inserts) gives the best acoustic performance if handled safely and encapsulated.

## Internal Bracing & Load Path (off-centre feet)

This design assumes the monitor foot pockets will be positioned slightly offset toward the outside of the monitor footprint (i.e. feet sit inward from the outer edge of the enclosure but not centered). Because of this, the enclosure must provide a direct vertical load path from each foot down into the bottom panel without relying on thin top-panel bending.

Key bracing elements:

- **Top doubler / foot pocket rib:** a local doubler under each foot pocket that transfers load straight down. The doubler should be glued and screwed to the top plate and tie into the nearest vertical brace.
- **Full-height centre brace:** a full-height vertical brace that spans top-to-bottom at or near the centre of the enclosure. This brace provides shear stiffness and prevents the side panels from parallelogramming.
- **Vertical foot-support braces:** two vertical braces directly beneath the foot pocket locations (these can be narrower than the centre brace) that run from the top plate to the bottom plate, creating a direct load column under each foot.
- **Bottom load-spreading plate / tray:** a thick bottom plate or tray that receives the vertical loads from the foot-support braces and also contains the ballast. The ballast should sit on the bottom tray, which is rigidly attached to the side walls and braces.
- **Horizontal stiffeners / mid-level brace:** one horizontal brace at approximately 1/3 height to add torsional stiffness and provide mounting for tweeter or crossover if needed.

Joinery & fasteners:

- Glue and screws for all major joints. Countersink screws and use glue in long-grain joints.
- Use hardwood dowels or biscuits for alignment on the top doubler and full-height brace if routing is available.
- Seal and fillet inside corners to avoid small vibrational pockets.

Layout notes:

- Keep heavy electronics and power supplies in a separate compartment at the rear/top of the enclosure if present, but try to keep heavy PCB transformers or battery packs low in the rear electronics compartment to avoid raising the COM.
- Place the bass reflex port in the front/top area as planned, ensuring the ballast does not obstruct internal port airflow.
- If the monitor feet are moved significantly inboard (>50 mm from the outer edge), increase the thickness of the top doubler or add an additional short pillar brace directly under the pocket.
